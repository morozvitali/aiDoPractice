    public List<String> myFilter3 () {
        List<String> words = List.of("APPLE", "Banana", "CHERRY", "kiwi", "PLUM");
        return words.stream().filter(a->a.length()%2 == 0)

    }

    public Map <Character, List<String>> myFilter4 () {

    }
